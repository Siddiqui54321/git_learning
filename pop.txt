#Create new branch
>git branch dev			--create new branch
>git checkout dev		--witch to the new branch
>git switch dev 		--switch to the new branch
>git branch			--check branch
>git checkout -b dev		--Create & switch in ONE command (shortcut)
>git switch -c dev		--Create & switch in ONE command (shortcut)
>git push -u origin dev		--Push branch to GitHub (important)
				--After creating the branch locally, upload it:
>git switch main		--Switch between branches
>git switch dev			--Switch between branches
